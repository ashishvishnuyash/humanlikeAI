# Phase 2 — Schema & Alembic Initial Migration

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Define all 22 SQLAlchemy models (matching Section 4 of the spec) and apply the initial Alembic migration to Azure Postgres. End state: every migration-target table exists in the Azure database, with indices and FKs, ready to receive data in Phase 6.

**Architecture:** One model file per domain (`user`, `company`, `mental_health`, `physical_health`, `community`, `calls`, `imports`). All inherit from `db.models.Base`. `users.id` is TEXT (preserves Firebase UIDs); all other primary keys are `UUID` with Python-side `uuid4` defaults. Freeform maps use `JSONB`; timestamp columns use `TIMESTAMPTZ` with server-side `NOW()` defaults. Alembic is initialized under `alembic/`, `env.py` wired to `Base.metadata`, initial migration generated and applied to `diltakdb.postgres.database.azure.com`.

**Tech Stack:** SQLAlchemy 2.0 declarative (`Mapped`, `mapped_column`), Alembic, Postgres 17.9, `sqlalchemy.dialects.postgresql` (UUID, JSONB, ARRAY).

**Spec reference:** `docs/superpowers/specs/2026-04-22-postgres-migration-design.md` — Section 4 (schema) and Section 8, Phase 2.

## File Structure

| Path | Responsibility |
|---|---|
| `alembic.ini` | Alembic CLI config (points to `alembic/`, reads `DATABASE_URL` from env) |
| `alembic/env.py` | Alembic migration runtime — wired to `Base.metadata` and `DATABASE_URL` |
| `alembic/script.py.mako` | Default template (generated by `alembic init`) |
| `alembic/versions/` | Auto-generated migration scripts go here |
| `db/models/__init__.py` | Exports `Base` and re-imports every model so `Base.metadata` is complete |
| `db/models/company.py` | `Company` |
| `db/models/user.py` | `User`, `RefreshToken` |
| `db/models/mental_health.py` | `CheckIn`, `Session`, `MentalHealthReport`, `ChatSession`, `AIRecommendation`, `Intervention`, `EscalationTicket` |
| `db/models/physical_health.py` | `PhysicalHealthCheckin`, `PhysicalHealthReport`, `MedicalDocument`, `WellnessEvent` |
| `db/models/community.py` | `CommunityPost`, `CommunityReply`, `AnonymousProfile`, `UserGamification`, `WellnessChallenge` |
| `db/models/calls.py` | `Call`, `CallSession` |
| `db/models/imports.py` | `ImportJob` |

**Import-order rule:** Companies exist before Users (users.company_id FK → companies.id). Users exist before everything else. The `__init__.py` imports in this order.

**Convention summary (applied uniformly):**
- Primary keys: `users.id = TEXT` (Firebase UID preserved); everything else `UUID` (Python `default=uuid.uuid4`, `as_uuid=True`).
- Foreign keys: explicit `ondelete` behavior on every FK, matching the spec.
- Timestamps: `created_at = TIMESTAMPTZ NOT NULL DEFAULT NOW()`; `updated_at = TIMESTAMPTZ NOT NULL DEFAULT NOW() ON UPDATE NOW()` (SQLAlchemy `onupdate=func.now()`).
- Freeform maps: `JSONB NOT NULL DEFAULT '{}'::jsonb`.
- String arrays: `ARRAY(TEXT) NOT NULL DEFAULT '{}'`.
- Indices: named explicitly (`ix_<table>_<cols>`) so Alembic autogen is stable.

```python
"""Alembic migration environment.

Reads ``DATABASE_URL`` from the environment (loaded by ``dotenv`` at import time
via ``db.session``) and uses ``db.models.Base.metadata`` as the target metadata.
"""

from __future__ import annotations

import os
from logging.config import fileConfig

from alembic import context
from sqlalchemy import engine_from_config, pool

# Load .env side-effect: importing db.session calls dotenv.load_dotenv().
from db.session import get_engine  # noqa: F401 — triggers env load
from db import models  # noqa: F401 — side-effect import so Base.metadata is populated
from db.models import Base

config = context.config

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata

_DATABASE_URL = os.environ.get("DATABASE_URL")
if not _DATABASE_URL:
    raise RuntimeError("DATABASE_URL is not set in the environment / .env")

# Inject the URL so both online and offline modes see it.
config.set_main_option("sqlalchemy.url", _DATABASE_URL)


def run_migrations_offline() -> None:
    """Emit SQL without connecting to the DB."""
    context.configure(
        url=_DATABASE_URL,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
        compare_type=True,
        compare_server_default=True,
    )
    with context.begin_transaction():
        context.run_migrations()


def run_migrations_online() -> None:
    """Connect to the DB and apply migrations."""
    connectable = engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
            compare_type=True,
            compare_server_default=True,
        )
        with context.begin_transaction():
            context.run_migrations()


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```

```python
"""User and refresh-token models.

``User.id`` is ``TEXT`` (not UUID) to preserve Firebase UIDs during migration.
New users created post-migration get a UUID string.
"""

from __future__ import annotations

import uuid
from datetime import datetime

from sqlalchemy import Boolean, ForeignKey, Index, String, func
from sqlalchemy.dialects.postgresql import JSONB, UUID
from sqlalchemy.orm import Mapped, mapped_column

from db.models import Base


class User(Base):
    __tablename__ = "users"

    id: Mapped[str] = mapped_column(String, primary_key=True)
    email: Mapped[str] = mapped_column(String, nullable=False, unique=True)
    password_hash: Mapped[str | None] = mapped_column(String, nullable=True)
    role: Mapped[str] = mapped_column(String, nullable=False)
    company_id: Mapped[uuid.UUID | None] = mapped_column(
        UUID(as_uuid=True),
        ForeignKey("companies.id", ondelete="SET NULL"),
        nullable=True,
    )
    manager_id: Mapped[str | None] = mapped_column(
        String,
        ForeignKey("users.id", ondelete="SET NULL"),
        nullable=True,
    )
    department: Mapped[str | None] = mapped_column(String, nullable=True)
    is_active: Mapped[bool] = mapped_column(
        Boolean, nullable=False, server_default="true"
    )
    profile: Mapped[dict] = mapped_column(
        JSONB, nullable=False, server_default="{}"
    )
    created_at: Mapped[datetime] = mapped_column(
        nullable=False, server_default=func.now()
    )
    updated_at: Mapped[datetime] = mapped_column(
        nullable=False, server_default=func.now(), onupdate=func.now()
    )

    __table_args__ = (
        Index("ix_users_company_active", "company_id", "is_active"),
        Index(
            "ix_users_company_active_dept",
            "company_id",
            "is_active",
            "department",
        ),
        Index("ix_users_manager_id", "manager_id"),
        Index("ix_users_role", "role"),
    )


class RefreshToken(Base):
    __tablename__ = "refresh_tokens"

    id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True), primary_key=True, default=uuid.uuid4
    )
    user_id: Mapped[str] = mapped_column(
        String,
        ForeignKey("users.id", ondelete="CASCADE"),
        nullable=False,
    )
    token_hash: Mapped[str] = mapped_column(String, nullable=False)
    expires_at: Mapped[datetime] = mapped_column(nullable=False)
    revoked: Mapped[bool] = mapped_column(
        Boolean, nullable=False, server_default="false"
    )
    created_at: Mapped[datetime] = mapped_column(
        nullable=False, server_default=func.now()
    )

    __table_args__ = (
        Index("ix_refresh_tokens_user_revoked", "user_id", "revoked"),
    )
```

```python
"""Verify the initial schema landed in Postgres.

Prints every table name, column, and index found in the public schema.
Exits 0 if all 22 expected tables exist, 1 otherwise.
"""

from __future__ import annotations

import sys

from sqlalchemy import inspect

from db.session import get_engine

EXPECTED_TABLES = {
    "ai_recommendations",
    "anonymous_profiles",
    "call_sessions",
    "calls",
    "chat_sessions",
    "check_ins",
    "community_posts",
    "community_replies",
    "companies",
    "escalation_tickets",
    "import_jobs",
    "interventions",
    "medical_documents",
    "mental_health_reports",
    "physical_health_checkins",
    "physical_health_reports",
    "refresh_tokens",
    "sessions",
    "user_gamification",
    "users",
    "wellness_challenges",
    "wellness_events",
}


def main() -> int:
    inspector = inspect(get_engine())
    tables = set(inspector.get_table_names(schema="public"))
    missing = EXPECTED_TABLES - tables
    extra = tables - EXPECTED_TABLES - {"alembic_version"}
    print(f"Found {len(tables)} tables in public schema.")
    if missing:
        print(f"MISSING: {sorted(missing)}", file=sys.stderr)
    if extra:
        print(f"UNEXPECTED: {sorted(extra)}", file=sys.stderr)
    if missing:
        return 1
    print("OK: all 22 expected tables present.")
    print("")
    for t in sorted(EXPECTED_TABLES):
        cols = [c["name"] for c in inspector.get_columns(t)]
        idxs = [i["name"] for i in inspector.get_indexes(t)]
        print(f"  {t}: {len(cols)} cols, {len(idxs)} idxs")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
